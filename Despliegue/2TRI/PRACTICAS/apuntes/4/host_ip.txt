default.conf:
<VirtualHost *:80>
        # The ServerName directive sets the request scheme, hostname and port that
        # the server uses to identify itself. This is used when creating
        # redirection URLs. In the context of virtual hosts, the ServerName
        # specifies what hostname must appear in the request's Host: header to
        # match this virtual host. For the default virtual host (this file) this
        # value is not decisive as it is used as a last resort host regardless.
        # However, you must set it for any further virtual host explicitly.
        #ServerName www.example.com

        ServerAdmin webmaster@empresadaw.com
        ServerName www.empresadaw.com
        ServerAlias www.empresadaw.local
        DocumentRoot /var/www/html/empresadaw

        ErrorLog ${APACHE_LOG_DIR}/empresadaw_error.log
        CustomLog ${APACHE_LOG_DIR}/empresadaw_access.log combined


        <Directory /var/www/html/empresadaw>
        Options FollowSymLinks MultiViews
        AllowOverride All
        Require all granted
        </Directory>
